ESTRUCTURA SENTENCIA BUCLE LOOP + EXCEPTION:

DECLARE
    contador NUMBER := 1;
    exception_prueba EXCEPTION;
BEGIN 
    LOOP
        BEGIN
            DBMS_OUTPUT.PUT_LINE(contador);
            IF contador = 20 THEN
                EXIT;
            END IF;
            
            contador := contador + 1;
        EXCEPTION
            WHEN OTHERS THEN
                DBMS_OUTPUT.PUT_LINE('Se ha producido un error en la variable');
        END;
    END LOOP;
END;

    DECLARE

        new_persona.edad := 72;
        -- Insertar los datos
            INSERT INTO persona VALUES new_persona;

        END; 


CREACION DE TABLA E INSERCION DE ELEMENTOS:

DECLARE 
-- Crear tipo de dato en la tabla
TYPE numero IS TABLE OF NUMBER;
-- Crear tabla e inicializar tabla con el tipo de dato
tabla_numeros numero := numero();
-- Variable para ver resultado la cosulta
resultado NUMBER;

BEGIN
FOR n IN 1..20 LOOP
    tabla_numeros.EXTEND;
    tabla_numeros(n) := n;
END LOOP;
resultado := tabla_numeros.COUNT;
DBMS_OUTPUT.PUT_LINE('Hay ' || resultado || ' nuemros en la tabla');
END;


DECLARACIÓN DE UN CURSOR:
DECLARE
    CURSOR cursor_empleados IS
        SELECT nombre
        FROM empleados;
    nombre_empleado empleados.nombre%TYPE; -- Declaración de la variable para almacenar el nombre del empleado
BEGIN
    OPEN cursor_empleados; -- Abrir el cursor

    -- Recorrido de los registros del cursor
    LOOP
        FETCH cursor_empleados INTO nombre_empleado;
        EXIT WHEN cursor_empleados%NOTFOUND; -- Salir del bucle cuando no hay más filas
        DBMS_OUTPUT.PUT_LINE('Empleado: ' || nombre_empleado);
    END LOOP;
